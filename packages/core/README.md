# @prettui/core

Core ANSI, chart, and layout primitives for prettui terminal interfaces.

```bash
npm install @prettui/core
```

```ts
import { fitAnsi, visibleLength } from "@prettui/core/ansi";
import { bars, segmentedDonut } from "@prettui/core/charts";
import { box, normalizeFrame } from "@prettui/core/layout";
```
